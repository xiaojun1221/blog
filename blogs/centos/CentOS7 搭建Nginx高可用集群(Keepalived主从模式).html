                  #!/bin/bash
path=$(pwd)

if [ ! -f ../soft/nginx-1.19.5.tar.gz ];then

nginx
echo 开放端口
firewall-cmd --zone=public --add-port=80/tcp --permanent
firewall-cmd --reload
echo 开启成功
cd $path

                  # 四、什么是keepalived

                  • keepalived：基于虚拟路由冗余协议(VRRP)，能够虚拟出一个IP对外暴露，当有节点出问题时，会进行选举操作进行故障转移(主从模式)

                  # 五、使用keepalived使多个节点自动进行故障转移

                  安装keepalive ，如下

                  # 5.1安装包

                  http://www.keepalived.org/software/keepalived-2.0.20.tar.gz tar zxvf keepalived-2.0.20.tar.gz

                  # 5.2、编译安装

                  解压 tar -xzvf keepalived-2.0.16.tar.gz cd keepalived-2.0.16 执行配置，指定路径 ./configure --prefix=/usr/local/keepalived 编译 make make install 拷贝源码中的keepalived-2.0.20/keepalived/etc/init.d/keepalived 到 /etc/init.d/ cp /usr/local/keepalived-2.0.20/keepalived/etc/init.d/keepalived /etc/init.d/ 拷贝编译后的keepalived/etc/sysconfig/keepalived 到 /etc/sysconfig/ cp /usr/local/keepalived/etc/sysconfig/keepalived /etc/sysconfig/ 创建文件夹 mkdir /etc/keepalived 把配置文件移到文件夹中 cp /usr/local/keepalived/etc/keepalived/keepalived.conf /etc/keepalived/ 把执行文件移动到/usr/sbin/ cp /usr/local/keepalived/sbin/keepalived /usr/sbin/

                  # 5.3、将keepalived添加到开机启动服务中，并进行测试

                  chkconfig keepalived on chkconfig --list | grep keepalived sudo service keepalived restart

                  **注意：**需要修改keepalived启动文件，否则可能会出现stop命令无法停止的情况(注释掉killMode这一行)

                  [root@ng2 system]# vim /usr/lib/systemd/system/keepalived.service
[Unit]
Description=LVS and VRRP High Availability Monitor
After=syslog.target network-online.target
[Service]
Type=forking
PIDFile=/var/run/keepalived.pid
#注释掉killMode这一行
#KillMode=process
EnvironmentFile=-/etc/sysconfig/keepalived
ExecStart=/usr/sbin/keepalived $KEEPALIVED_OPTIONS
ExecReload=/bin/kill -HUP $MAINPID
[Install]
WantedBy=multi-user.target

                  # 六、配置检查脚本使keepalived可以感知nginx的状态，并做故障转移

                  • 上面的操作，虽然实现了故障转移，但是是基于keepalived故障的，也就是说只有keepalived挂掉，才会转移到备机上，而我们需要的是nginx挂掉以后就做转移，而不是keepalived

                  • 编写nginx检测脚本

                  [root@ng2 script]# vim /usr/local/src/check_nginx_pid.sh

#!/bin/bash
#获取nginx进程的数量
counter=$(ps -C nginx --no-heading|wc -l)
#如果nginx进程数量为0(没有启动或挂掉)
echo monitor running
if [ ${counter} = 0 ]; then
        #则尝试启动一下nginx
    systemctl start nginx.service
    #等待2秒
    sleep 2
    #再查询nginx进程数量
    counter=$(ps -C nginx --no-heading|wc -l)
    #如果nginx进程数仍然为0
    if [ ${counter} = 0 ]; then
        #则停止keepalived进程，迫使虚拟IP转移到备机上，完成故障转移（该命令需要根据实际情况修改）
        systemctl stop keepalived.service
    fi
fi


#设置为为可执行文件
[root@ng2 src]# chmod +x check_nginx_pid.sh

                  # 七、主从模式总结

                  整个过程为：

                  1. 首先得有两个或多个一模一样的nginx节点
                  2. 每个节点上安装keepalived，并且使用同一个虚拟ip(暴露给应用服务使用)，这些节点上，只有一个节点会有虚拟IP，当访问时也就访问到这个节点上。
                  3. keepalived每次会定时运行监测脚本
                  4. 如果keepalived挂掉或被停止掉，会将虚拟IP转移至某一个从节点使其来提供服务，所以第3步就是利用这一点，发现nginx故障时，停止掉keepalived，触发故障转移，同理可以通过改写脚本，几乎可以对任何应用做主从的高可用
                  5. 可以配置警告邮箱，当有故障发生时，可以获得通知，并处理故障。如果故障不处理，当所有节点都挂掉后，服务也就不可用了。
                  6. 还可以做双主模式，但是这样会有两个虚拟IP出现，不便统一暴露给应用程序
                  7. 以上是基本的高可用配置
                    • 其它配置说明参考：https://github.com/huweihuang/linux-notes/blob/master/keepalived/keepalived-conf.md
                    • 更多的配置可以参考官方配置说明(英文)：https://www.keepalived.org/manpage.html
