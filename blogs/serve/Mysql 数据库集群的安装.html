# 1 两台服务器安装mysql,脚本如下

#!/bin/bash
path=$(pwd)
if [ ! -f ../soft/mysql-5.7.26-linux-glibc2.12-x86_64.tar.gz ];then
   echo mysql安装包不存在----下载安装包....
   wget ../soft/mysql-5.7.26-linux-glibc2.12-x86_64.tar.gz http://study.zyjblogs.cn/software/mysql-5.7.26-linux-glibc2.12-x86_64.tar.gz
   echo 下载完成
fi
echo 执行解压
tar -zxvf ../soft/mysql-5.7.26-linux-glibc2.12-x86_64.tar.gz
echo 移动mysql目录到/usr/local/mysql
if [  -d /usr/local/mysql ]; then
       rm -rf /usr/local/mysql
fi
mv mysql-5.7.26-linux-glibc2.12-x86_64  /usr/local/mysql
echo 移动完成
echo 开始创建mysql用户
#创建组
groupadd mysql
#给组添加用户
useradd -r -g mysql mysql
echo 创建用户完成
echo 开始创建目录
mkdir -p  /usr/local/mysql/data
mkdir /usr/local/mysql/log/
touch /usr/local/mysql/log/error.log
echo 赋予权限
chown mysql:mysql -R /usr/local/mysql
echo 配置文件
cp ../conf/mysql/my.cnf /etc/
echo 配置 配置文件完成
echo 初始化数据库
#初始化数据库
/usr/local/mysql/bin/mysqld --defaults-file=/etc/my.cnf --basedir=/usr/local/mysql/ --datadir=/usr/local/mysql/data/ --user=mysql --initialize
#复制
cp /usr/local/mysql/support-files/mysql.server /etc/init.d/mysql
#查看密码
cat /usr/local/mysql/log/error.log
#启动服务
service mysql start
echo 开启防火墙
firewall-cmd --zone=public --add-port=3306/tcp --permanent
firewall-cmd --reload
echo 开启成功
cd $path

# 9 Nginx配置

Nginx配置MySQL集群访问URL，确保微服务应用连接相同的URL。 Nginx中的MySQL配置，内容如下:

stream {
  upstream mysql_proxy{
    hash $remote_addr consistent;
    server A服务器IP:3306 weight=1 max_fails=3 fail_timeout=10s;
      server B服务器IP:3306 weight=1 max_fails=3 fail_timeout=10s;
  }
  server {
    listen 3306; # 数据库服务器监听端口
    proxy_connect_timeout 10s;
    proxy_timeout 300s; 
    proxy_pass mysql_proxy;
  }
}
